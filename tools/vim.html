# 命令模式

i # 进入编辑模式
w # 保存
q # 退出
! # 强制
wq a.py # 指定名字为a.py保存
syntax on # 打开语法高亮
set number # 显示行号
cc # 剪切当前行
pp # 粘贴到当前行
u # 撤销
yy # 复制当前行
dd # 删除当前行
ctrl + n # 自动补全
o # 在当前行之下新加一行，并进入插入模式
O # 在肖前行之上新加一行，并进行插入模式
