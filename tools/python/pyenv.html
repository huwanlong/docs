# 安装配置

# 安装

pip install pyenv-win --target %USERPROFILE%/.pyenv
1

# 环境变量

设置一个系统变量PYENV，值为C:\Users\用户名\.pyenv\pyenv-win

path 里面添加：

%PYENV%\bin
%PYENV%\shims
1
2

# 换源

将C:\Users\用户名\.pyenv\pyenv-win\.versions_cache.xml中 https://www.python.org/ftp/python 全换成 https://npm.taobao.org/mirrors/python

# 常用命令

# 列出当前系统中所有安装的python
pyenv versions
# 显示出当前使用的python
pyenv version
# 设置使用哪一个python
pyenv global <python_version>
# 恢复系统版本
pyenv global system
# 通过将版本号写入当前目录下的 .python-version 文件的方式设置 Python 本地版本
pyenv local <python_version>
# 取消本地版本设置
pyenv local –unset
# 安装特定版本的python
pyenv install <python_version> # pyenv install 3.8.7
# 移除特定版本的python
pyenv uninstall <python_version>
# 查看可安装的python
pyenv install -l
# 升级
pyenv update
# 重建环境变量，每当你增删python版本或增删带可执行性的包（如pip）以后，都应该执行一次本命令
pyenv rehash

# 使用方法

# 可以将用pyenv安装的python目录配置到环境变量中，
# 或是使用pyenv global XXX 切换到一个特定的python版本
# 然后进入项目目录中，创建虚拟环境
python -m venv myvirtuaenv # python -m venv XXX（虚拟环境名）
# 激活虚拟环境
.\myvirtuaenv\Scripts\activate
# 退出虚拟环境
deactivate
